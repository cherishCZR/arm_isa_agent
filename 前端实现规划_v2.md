# ARM ISA Compiler Verification Agent — 前端实现规划 v2

> **修订要点**：本版重点覆盖 (1) 测试程序下载（单条/批量）(2) 推理过程可视化 (3) 对话模式复制/下载双通道。

---

## 1. 当前系统缺口分析

基于对 `orchestrator.py`、`models.py`、`schemas.py`、`generation/models.py` 的完整阅读，当前系统存在以下不足：

| # | 缺口 | 现状 | 影响 |
|---|------|------|------|
| 1 | **完整测试程序不可获取** | `test_suite_preview` 截断到 2000 字符，`TestCaseSuite` 对象（含 8 种格式、`.to_assembly_file()` / `.to_cpp_program()` 等方法）在管道结束后丢弃 | 用户无法下载完整 `.s`/`.c`/`.cpp` 文件 |
| 2 | **推理过程不可见** | `StageResult.output_summary` 仅为简短字符串（如 `"Generated 20 test cases across 5 formats"`），不包含约束分析详情、规划推理链 | 前端只能展示结果，无法展示"为什么这样生成" |
| 3 | **无按格式下载端点** | API 没有任何 test file download 端点 | 前端无法提供 `.s`/`.c`/`.cpp` 格式选择 |
| 4 | **对话模式输出无结构** | Agent chat 的 `final_answer` 是 LLM 生成的纯 Markdown 文本，不包含结构化 test file 数据 | 对话中生成的测试程序无法被精确提取用于下载 |
| 5 | **无逐测试用例通过/失败标记** | 审查只给出整体 score，不标记具体哪些测试有问题、哪些无问题 | 无法筛选"验证通过的测试程序"供下载 |

## 2. 后端改造计划

### 2.1 新增数据模型

```python
# verification/models.py — 新增

class TestFileEntry(BaseModel):
    """单个可下载的测试文件。"""
    file_id: str                     # 唯一标识，如 "ADD_asm_001"
    filename: str                    # 建议文件名，如 "ADD_test_normal_01.s"
    format: Literal["s", "c", "cpp", "llvm"]  # 文件格式
    test_type: str                   # assembly | inline_asm | cpp_verification | llvm_mc | boundary | alias | invalid | feature
    content: str                     # 完整文件内容
    status: Literal["pass", "fail"]  # 逐测试用例的审查结果
    description: str                 # 测试描述
    issue_count: int = 0             # 该测试关联的 issue 数

class StageDetail(BaseModel):
    """增强的阶段详情，含推理链。"""
    stage: str
    status: str
    duration_ms: float
    summary: str                     # 一句话摘要
    reasoning: str = ""              # 详细推理过程
    key_findings: list[str] = []     # 关键发现点
    data_snapshot: dict = {}         # 阶段产出数据快照
        # retrieval:         {encoding_count, operand_count, complexity, ...}
        # constraint_analysis: {unpredictable_count, feature_deps, immediate_ranges, ...}
        # test_planning:     {dimensions, total_tests, coverage_matrix, reasoning, ...}
        # test_generation:   {format_counts, generation_mode, warnings, ...}
        # test_review:       {dimension_scores, issues_by_severity, ...}
        # repair:            {repair_count, changes, ...}
```

### 2.3 Orchestrator 改造 (关键)

`VerificationOrchestrator.verify()` 需要以下改造：

```python
# orchestrator.py 改造要点

# ── Stage 1: Retrieval — 丰富 output → StageDetail ──
# 原来:
#   output_summary = f"Loaded profile: {profile.mnemonic}, class=..."
# 改造为:
stage_results.append(StageDetail(
    stage="retrieval",
    status="ok",
    duration_ms=...,

        f"操作数: {profile.operand_count} 个 ({', '.join(o.symbol for o in profile.operands[:5])})",
        f"特性依赖: {len(profile.feature_dependencies)} 个",

    data_snapshot={
        "mnemonic": profile.mnemonic,
        "instr_class": profile.instr_class,
        "encoding_count": profile.encoding_count,
        "operand_count": profile.operand_count,
        "complexity": profile.complexity_score,

        "affects_flags": profile.affects_flags,

# ── Stage 2: Constraint Analysis — 展示约束详情 ──
stage_results.append(StageDetail(
    stage="constraint_analysis",
    status="ok",
    duration_ms=...,
    summary=f"发现 {len(constraints)} 条约束",
    reasoning=f"分析指令 {profile.mnemonic} 的操作数约束..."
             f"检测到 {len(immediate_ranges)} 个立即数字段..."
             f"寄存器分析: GP×{gp_count}, SIMD×{simd_count}...",
    key_findings=constraint_details,  # 逐条约束
    data_snapshot={
        "constrained_unpredictable": [...],  # 每条约束的条件和描述
        "immediate_ranges": [...],           # 每个立即数的位宽/范围
        "feature_gates": [...],
        "encoding_undefined": [...],
        "register_widths": [...],
    },
))

# ── Stage 3: Test Planning — 展示完整推理链 ──
stage_results.append(StageDetail(
    stage="test_planning",
    status="ok",
    duration_ms=...,
    summary=f"规划 {total_tests} 个测试用例，{len(dims)} 个维度",
    reasoning=plan_output.reasoning if plan_output else "",
    key_findings=[
        f"维度 {d.name}: 优先级={d.priority}, 建议 {d.suggested_test_count} 个用例"
        for d in strategy.dimensions
    ] if strategy else [],
    data_snapshot={
        "dimensions": [d.model_dump() for d in strategy.dimensions] if strategy else [],
        "total_test_count": strategy.total_test_count if strategy else 0,
        "coverage_matrix": strategy.coverage_matrix if strategy else {},
        "risk_analysis": strategy.risk_analysis if strategy else "",
    },
))

# ── Stage 4-6: 同逻辑增强 ──

# ── 构建 TestFileEntry 列表 ──
test_files: list[TestFileEntry] = []
if suite:
    # 8 类测试 → 各自的文件格式
    for i, t in enumerate(suite.assembly_tests):
        test_files.append(TestFileEntry(
            file_id=f"{instruction}_asm_{i:03d}",
            filename=f"{instruction}_{t.test_name}.s",
            format="s",
            test_type="assembly",
            content=t.to_assembly_file(),
            status="pass",  # 默认 pass，审查时更新
            description=t.description,
        ))
    for i, t in enumerate(suite.llvm_mc_tests):
        test_files.append(TestFileEntry(
            file_id=f"{instruction}_llvm_{i:03d}",
            filename=f"{instruction}_{t.test_name}.s",
            format="llvm",
            test_type="llvm_mc",
            content=t.to_llvm_mc_file(),
            status="pass",
            description=t.description,
        ))
    for i, t in enumerate(suite.inline_asm_tests):
        test_files.append(TestFileEntry(
            file_id=f"{instruction}_inline_{i:03d}",
            filename=f"{instruction}_{t.test_name}.c",
            format="c",
            test_type="inline_asm",
            content=t.to_inline_asm_code(),
            status="pass",
            description=t.description,
        ))
    for i, t in enumerate(suite.cpp_verification_tests):
        test_files.append(TestFileEntry(
            file_id=f"{instruction}_cpp_{i:03d}",
            filename=f"{instruction}_{t.test_name}.cpp",
            format="cpp",
            test_type="cpp_verification",
            content=t.to_cpp_program(),
            status="pass",
            description=t.description,
        ))
    # boundary, alias, invalid 转换为 assembly:
    for i, t in enumerate(suite.boundary_tests):
        asm = t.to_assembly_test()
        test_files.append(TestFileEntry(
            file_id=f"{instruction}_boundary_{i:03d}",
            filename=f"{instruction}_{t.test_name}.s",
            format="s",
            test_type="boundary",
            content=asm.to_assembly_file(),
            status="pass",
            description=f"[{t.category}] {t.boundary_value}",
        ))
    for i, t in enumerate(suite.alias_tests):
        asm = t.to_assembly_test()
        test_files.append(TestFileEntry(...))
    for i, t in enumerate(suite.invalid_operand_tests):
        asm = t.to_assembly_test()
        test_files.append(TestFileEntry(...))
    # feature enable → LLVM MC:
    for i, t in enumerate(suite.feature_enable_tests):
        llvm = t.to_llvm_mc_test()
        test_files.append(TestFileEntry(
            filename=f"{instruction}_{t.test_name}.s",
            format="llvm",
            content=llvm.to_llvm_mc_file(),
            ...
        ))

# ── 逐测试标记审查结果 ──
if review_result:
    for issue in review_result.issues:
        if issue.location:
            # 匹配 location（如 test_name）到 test_files
            for tf in test_files:
                if issue.location in tf.filename or issue.location in tf.description:
                    tf.status = "fail"
                    tf.issue_count += 1
```

### 2.4 新增 API 端点

| 端点 | 方法 | 用途 |
|------|------|------|
| `POST /api/generate_testcase` | POST | **改造** — 响应增加 `test_files`、`passing_test_count`、`failing_test_count`、`stage_results` 增强 |
| `POST /api/generate_testcases` | POST | **改造** — 批量响应增加每指令的 `test_files` |

### 2.5 对话模式改造

当前 Agent 输出是 LLM 生成的纯 Markdown 文本。改造方案：

**方案：在 `tool_results` 中注入结构化数据，Summarizer 同时输出文本 + 结构化 block**

```python
# AgentGraph.run() / arun() 返回增强
return {
    "answer": "...",                    # Markdown 文本（用于展示）
    "iterations": ...,
    "tool_results": [...],
    "review_results": [...],
    "verification_report": {...},
    # ── 新增 ──
    "test_files": [...],                # list[TestFileEntry] — 结构化测试文件
    "stage_details": [...],             # list[StageDetail] — 推理链
    "plan_reasoning": "...",            # 规划推理

```

## 3. 前端设计（修订版）

### 3.1 技术选型（不变）

```typescript
// lib/hooks/use-download.ts

import JSZip from "jszip";
import { saveAs } from "file-saver";

export function useDownload() {
  // 下载单个文件
  const downloadSingle = (entry: TestFileEntry, format?: string) => {
    const ext = format || entry.format;
    const filename = entry.filename.replace(/\.[^.]+$/, `.${ext}`);
    // 如果格式不匹配，从 content 中提取对应格式
    const content = format ? extractFormat(entry.content, format) : entry.content;
    const blob = new Blob([content], { type: "text/plain" });
    saveAs(blob, filename);
  };

  // 下载全部通过测试为 ZIP
  const downloadAllAsZip = async (
    instruction: string,
    entries: TestFileEntry[],
    onlyPassing = true
  ) => {
    const zip = new JSZip();
    const folder = zip.folder(instruction)!;

    const files = onlyPassing
      ? entries.filter(e => e.status === "pass")
      : entries;

    files.forEach(entry => {
      folder.file(entry.filename, entry.content);
    });

    const blob = await zip.generateAsync({ type: "blob" });
    saveAs(blob, `${instruction}_testcases.zip`);
  };

  // 批量下载多指令（ZIP 内分子目录）
  const batchDownloadAsZip = async (reports: VerifyResponse[]) => {
    const zip = new JSZip();

    reports.forEach(report => {
      const folder = zip.folder(report.instruction)!;
      const passing = report.test_files.filter(f => f.status === "pass");
      passing.forEach(entry => {
        folder.file(entry.filename, entry.content);
      });
    });

    const blob = await zip.generateAsync({ type: "blob" });
    const date = new Date().toISOString().slice(0, 10);
    saveAs(blob, `arm_testcases_${date}.zip`);
  };

  return { downloadSingle, downloadAllAsZip, batchDownloadAsZip };
}

// 从混合内容中提取特定格式
function extractFormat(content: string, format: string): string {
  // 如果原始格式匹配，直接返回
  // 否则尝试从 content 中解析出对应格式的代码块
  // 例如：从一个包含 asm + c 的 test_suite_full 中提取 asm 部分
  return content;
}
```

```typescript
export function CopyButton({ content, label = "复制" }: {
```

**前端 SSE 处理：**

```typescript
// lib/hooks/use-chat.ts

type ChatEvent =
  | { type: "thinking"; stage: string; message: string }
  | { type: "stage_complete"; stage: string; detail: StageDetail }
  | { type: "token"; content: string }
  | { type: "test_files"; files: TestFileEntry[]; passing: number; failing: number }
  | { type: "done"; answer: string; report?: VerifyResponse };

function useChatStream() {
  const [stages, setStages] = useState<StageDetail[]>([]);
  const [tokens, setTokens] = useState("");
  const [testFiles, setTestFiles] = useState<TestFileEntry[]>([]);
  const [currentStage, setCurrentStage] = useState<string | null>(null);
  const [isThinking, setIsThinking] = useState(false);

  const handleEvent = (event: ChatEvent) => {
    switch (event.type) {
      case "thinking":
        setCurrentStage(event.stage);
        setIsThinking(true);
        break;
      case "stage_complete":
        setStages(prev => [...prev, event.detail]);
        setCurrentStage(null);
        break;
      case "token":
        setTokens(prev => prev + event.content);
        break;
      case "test_files":
        setTestFiles(event.files);
        break;
      case "done":
        setIsThinking(false);
        break;
    }
  };

  return { stages, tokens, testFiles, currentStage, isThinking, handleEvent };
}
```

## 6. 对话框/下载行为决策表

| 场景 | 默认展示 | 下载按钮 | 文件格式 |
|------|----------|----------|----------|
| **对话中说"帮我生成 ADD 的测试"** | 代码块 + 📋复制 | 每文件下方 💾.s 💾.c 💾.cpp | 自动检测 |
| **对话中说"生成测试并下载 .s 文件"** | 代码块 + 📋复制 | 下载自动触发 .s | .s |
| **对话中说"生成 .cpp 和 .s 文件给我"** | 代码块 + 📋复制 | 下载自动触发 .cpp + .s | .cpp, .s |
| **验证页提交指令** | 结果摘要 + 推理时间线 + 文件列表 | 顶部 [💾 全部] | ZIP (含全部格式) |
| **批量验证完成** | 进度表 + 汇总 | [💾 全部通过] 按指令 [💾] | ZIP (分子目录) |
| **验证结果中筛选仅通过** | 仅通过文件列表 | [💾 所选] | ZIP |
| **点击单个文件** | 预览弹窗 + 📋 | 弹窗内 💾 | 单格式 |

**关键规则：**
1. 对话中未声明下载 → 优先展示可复制代码块，下载按钮作为辅助
2. 对话中声明了下载 → 自动触发下载 + 同时展示代码块
3. 验证页面 → 始终提供下载入口，用户可以主动选择
